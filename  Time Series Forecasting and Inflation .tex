# Time Series Forecasting & Predictive Modeling
# Python Code for Data Science Intern Application

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_absolute_percentage_error

# Generate synthetic time series data (simulates real-world data)
time = np.arange(1, 101)
trend = 0.5 * time
seasonal = 4 * np.sin(time * 0.1)
noise = np.random.normal(0, 1, len(time))
y = trend + seasonal + noise

# Create structured DataFrame
df = pd.DataFrame({
    'time': time,
    'value': y
})

# Build predictive model
model = LinearRegression()
model.fit(df[['time']], df['value'])

# Forecast next 20 steps
forecast_time = np.arange(101, 121)
forecast = model.predict(forecast_time.reshape(-1, 1))

# Visualize results
plt.figure(figsize=(12, 5))
plt.plot(df['time'], df['value'], label='Historical Data')
plt.plot(forecast_time, forecast, '--', label='Forecast', color='red')
plt.title('Time Series Forecasting Model')
plt.xlabel('Time')
plt.ylabel('Value')
plt.legend()
plt.show()

# Model performance
y_pred = model.predict(df[['time']])
mape = mean_absolute_percentage_error(y, y_pred)
print(f'Model MAPE: {mape:.4f}')
print('Time series forecasting completed successfully.')
